{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "portfolio.fullname" . }} -o jsonpath="{.spec.ports[0].nodePort}")
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "  http://$NODE_IP:$NODE_PORT"

{{- else if eq .Values.service.type "LoadBalancer" }}
  It may take a few minutes for the LoadBalancer IP to be provisioned.
  Watch with:
    kubectl get svc --namespace {{ .Release.Namespace }} {{ include "portfolio.fullname" . }} -w

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "portfolio.fullname" . }} --template "{{ "{{" }}range (index .status.loadBalancer.ingress 0){{ "}}" }}{{ "{{" }}.{{ "}}" }}{{ "{{" }}end{{ "}}" }}")
  echo "  http://$SERVICE_IP:{{ .Values.service.port }}"

{{- else }}
  # Forward a local port (quick access without Ingress):
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "portfolio.fullname" . }} 8080:{{ .Values.service.port }}

  Open: http://localhost:8080
{{- end }}
